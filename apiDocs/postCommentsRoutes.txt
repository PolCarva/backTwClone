// POST COMMENTS ROUTES

/:postid/crearcomentario
post:
summary: se crea un comentario para el post definido por parametros
responses:
        200:
 *      description: success: true, message: 'comentario creado'.
 *      content:
 *        application/json:
 *          schema:
 *            type: object
 *            properties:
 *              params:
 *                id del post
 *              data:
 *                type: object
 *                properties:
 *                  comment:
 *                    type: string
 *                    description: el texto del comentario.
 *                    example: 'este es mi primer comentario'                     
 
/:postid/comentarios
 get:
 summary: devuelve todos los comentarios del post definido por parametros
 responses:
         200:
  *      description: success: true, data: [postComments].
  *      content:
  *        application/json:
  *          schema:
  *            type: object
  *            properties:
  *            params:
  *                id del post
  *              data:
  *                type: object
  *                properties:
  *                  comment:
  *                    type: string
  *                    description: comentario.
  *                    example: 'mi primer comentario'                   


 /:postid/comentarios/:commentid
 get:
 summary: devuelve en detalle un comentario de un post
 responses:
       200:
  *      description: success: true, data: postComment.
  *      content:
  *        application/json:
  *          schema:
  *            type: object
  *            properties:
  *              params:
  *               id del post,
  *               id del comentario      
  *              data:
  *                type: object  
  *                  comment:
  *                    type: string
  *                    description: comentario del post.
  *                    example: 'mi primer comentario' 
  *                  replies:
  *                    type: [string]
  *                    description: respuestas al comentario.
